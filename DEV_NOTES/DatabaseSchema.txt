-- ============================================
-- ACADEMIC YEAR
-- ============================================
CREATE TABLE academic_year (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    beginning_year INTEGER NOT NULL,
    ending_year INTEGER NOT NULL,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- ROLES
-- ============================================
CREATE TABLE roles (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    role VARCHAR(50) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- DEPARTMENTS
-- ============================================
CREATE TABLE departments (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- SECTIONS
-- ============================================
CREATE TABLE sections (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- USERS
-- ============================================
CREATE TABLE users (
    user_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    role_id INTEGER NOT NULL,
    department_id INTEGER,
    section_id INTEGER,
    password TEXT NOT NULL,
    operator_type VARCHAR(20),
    access_level VARCHAR(30),
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    active BOOLEAN DEFAULT TRUE,
    CONSTRAINT fk_users_role
        FOREIGN KEY (role_id)
        REFERENCES roles (id)
        ON DELETE RESTRICT,
    CONSTRAINT fk_users_department
        FOREIGN KEY (department_id)
        REFERENCES departments (id)
        ON DELETE SET NULL,
    CONSTRAINT fk_users_section
        FOREIGN KEY (section_id)
        REFERENCES sections (id)
        ON DELETE SET NULL,
    CONSTRAINT chk_user_dept_or_section CHECK (
        (department_id IS NOT NULL AND section_id IS NULL)
        OR
        (department_id IS NULL AND section_id IS NOT NULL)
        OR
        (department_id IS NULL AND section_id IS NULL)
    )
);

-- ============================================
-- STUDENTS
-- ============================================
CREATE TABLE students (
    student_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
    roll_number VARCHAR(20) NOT NULL UNIQUE,
    branch VARCHAR(50) NOT NULL,
    section VARCHAR(10) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    mobile VARCHAR(15),
    father_name VARCHAR(100),
    father_mobile VARCHAR(15),
    academic_year_id INTEGER NOT NULL,
    role_id INTEGER NOT NULL,
    password TEXT NOT NULL,
    department_id INTEGER,
    section_id INTEGER,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_students_academic_year
        FOREIGN KEY (academic_year_id)
        REFERENCES academic_year (id)
        ON DELETE RESTRICT,
    CONSTRAINT fk_students_role
        FOREIGN KEY (role_id)
        REFERENCES roles (id)
        ON DELETE RESTRICT,
    CONSTRAINT fk_students_department
        FOREIGN KEY (department_id)
        REFERENCES departments (id)
        ON DELETE SET NULL,
    CONSTRAINT fk_students_section
        FOREIGN KEY (section_id)
        REFERENCES sections (id)
        ON DELETE SET NULL
);

-- ============================================
-- FACULTY
-- ============================================
CREATE TABLE faculty (
    faculty_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    employee_code VARCHAR(20) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL,
    department_id INTEGER,
    section_id INTEGER,
    designation TEXT[],
    email VARCHAR(100),
    mobile VARCHAR(15),
    role_id INTEGER NOT NULL,
    staff_type VARCHAR(50),
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    password TEXT NOT NULL,
    CONSTRAINT chk_department_or_section CHECK (
        (department_id IS NOT NULL AND section_id IS NULL)
        OR
        (department_id IS NULL AND section_id IS NOT NULL)
    ),
    CONSTRAINT fk_faculty_department
        FOREIGN KEY (department_id)
        REFERENCES departments (id)
        ON DELETE SET NULL,
    CONSTRAINT fk_faculty_section
        FOREIGN KEY (section_id)
        REFERENCES sections (id)
        ON DELETE SET NULL,
    CONSTRAINT fk_faculty_role
        FOREIGN KEY (role_id)
        REFERENCES roles (id)
        ON DELETE RESTRICT
);

-- ============================================
-- DUE TYPES
-- ============================================
CREATE TABLE due_types (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    type_name VARCHAR(100) NOT NULL,
    description TEXT,
    is_for_student BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    requires_permission BOOLEAN DEFAULT FALSE
);

-- ============================================
-- STUDENT DUES
-- ============================================
CREATE TABLE public.student_dues (
    -- Identification and Student Reference
    id INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY,
    student_roll_number CHARACTER VARYING(20) NOT NULL,
    
    -- Audit/Source Information
    added_by_user_id UUID NOT NULL,
    added_by_department_id INTEGER,
    added_by_section_id INTEGER,
    
    -- Due Specifics
    due_type_id INTEGER NOT NULL,
    due_description TEXT,
    is_payable BOOLEAN NOT NULL,
    
    -- Financials
    current_amount NUMERIC(10,2),
    amount_paid NUMERIC(10,2) DEFAULT 0,
    principal_amount NUMERIC(10,2),
    is_compounded BOOLEAN,
    interest_rate NUMERIC(10,6),
    
    -- Status and Permissions
    permission_granted BOOLEAN DEFAULT false,
    is_cleared BOOLEAN DEFAULT false,
    overall_status BOOLEAN DEFAULT false,
    cleared_by_user_id UUID,
    due_clear_by_date DATE NOT NULL,
    
    -- Documentation and Files
    supporting_document_link TEXT,
    proof_drive_link TEXT,
    needs_original BOOLEAN,
    needs_pdf BOOLEAN,
    remarks TEXT,
    
    -- Timestamps
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,

    -- Constraints
    CONSTRAINT student_dues_pkey PRIMARY KEY (id),
    
    -- Check Constraints
    CONSTRAINT chk_amount_logic CHECK (
        (is_payable = true AND current_amount IS NOT NULL) OR 
        (is_payable = false AND current_amount IS NULL)
    ),
    CONSTRAINT chk_amount_paid CHECK (amount_paid <= COALESCE(current_amount, 0::numeric)),
    CONSTRAINT chk_due_source CHECK (
        (added_by_department_id IS NOT NULL AND added_by_section_id IS NULL) OR 
        (added_by_department_id IS NULL AND added_by_section_id IS NOT NULL)
    ),
    CONSTRAINT chk_interest_rate_logic CHECK (
        (is_compounded = true AND interest_rate IS NOT NULL) OR 
        (is_compounded = false OR is_compounded IS NULL)
    ),
    CONSTRAINT chk_overall_status CHECK (
        overall_status = (COALESCE(is_cleared, false) OR COALESCE(permission_granted, false))
    ),

    -- Foreign Key Constraints
    CONSTRAINT fk_student_dues_cleared_by FOREIGN KEY (cleared_by_user_id) REFERENCES users(user_id) ON DELETE SET NULL,
    CONSTRAINT fk_student_dues_department FOREIGN KEY (added_by_department_id) REFERENCES departments(id) ON DELETE RESTRICT,
    CONSTRAINT fk_student_dues_due_type FOREIGN KEY (due_type_id) REFERENCES due_types(id) ON DELETE RESTRICT,
    CONSTRAINT fk_student_dues_section FOREIGN KEY (added_by_section_id) REFERENCES sections(id) ON DELETE RESTRICT,
    CONSTRAINT fk_student_dues_student FOREIGN KEY (student_roll_number) REFERENCES students(roll_number) ON DELETE CASCADE,
    CONSTRAINT fk_student_dues_user FOREIGN KEY (added_by_user_id) REFERENCES users(user_id) ON DELETE RESTRICT
);

-- Indexes
CREATE INDEX idx_student_dues_dept_cleared ON public.student_dues (student_roll_number, is_cleared, is_payable);
CREATE INDEX idx_student_dues_due_type ON public.student_dues (due_type_id);
CREATE INDEX idx_student_dues_student ON public.student_dues (student_roll_number);
CREATE INDEX idx_student_dues_user ON public.student_dues (added_by_user_id);

-- ============================================
-- DUE PAYMENTS
-- ============================================
CREATE TABLE due_payments (
    id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    due_id INTEGER NOT NULL,
    paid_amount NUMERIC(10,2) NOT NULL CHECK (paid_amount > 0),
    payment_reference VARCHAR(100) NOT NULL UNIQUE,
    payment_method VARCHAR(30) NOT NULL,
    payment_status payment_status_enum NOT NULL,
    gateway_response JSONB,
    paid_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_due_payments_due
        FOREIGN KEY (due_id)
        REFERENCES student_dues (id)
        ON DELETE CASCADE
);

-- ============================================
-- INDEXES
-- ============================================
CREATE INDEX idx_students_academic_year ON students (academic_year_id);
CREATE INDEX idx_students_role ON students (role_id);
CREATE INDEX idx_students_department_id ON students (department_id);
CREATE INDEX idx_students_section_id ON students (section_id);

CREATE INDEX idx_faculty_department ON faculty (department_id);
CREATE INDEX idx_faculty_section ON faculty (section_id);

CREATE INDEX idx_student_dues_student ON student_dues (student_roll_number);
CREATE INDEX idx_student_dues_user ON student_dues (added_by_user_id);
CREATE INDEX idx_student_dues_due_type ON student_dues (due_type_id);
CREATE INDEX idx_student_dues_documentation ON student_dues (needs_original, needs_pdf) WHERE is_payable = FALSE;

CREATE INDEX idx_due_payments_due ON due_payments (due_id);
